import data from '../eval-results/data.json';

const fileChanges = data.grade?.fileChanges ?? [];
const ghostStories = data.grade?.ghostStories;
const projectName = data.project?.name ?? '-';
const promptName = data.prompt?.name ?? '-';
const agent = data.variant?.agent ?? '-';
const model = data.variant?.model ?? '-';
const effort = data.variant?.effort ?? '-';
const score = data.score?.score ?? '-';
const build =
  data.grade?.buildSuccess === true ? 'PASS' : data.grade?.buildSuccess === false ? 'FAIL' : '-';
const typeCheckErrors = data.grade?.typeCheckErrors ?? '-';
const duration =
  typeof data.execution?.duration === 'number' ? `${Math.round(data.execution.duration)}s` : '-';
const cost =
  typeof data.execution?.cost === 'number' ? `$${data.execution.cost.toFixed(2)}` : '-';
const ghostStoriesSummary = ghostStories
  ? `${ghostStories.passed}/${ghostStories.total} (${Math.round(ghostStories.successRate * 100)}%)`
  : '-';
