import {
  changedFiles,
  emptySummaryText,
  ghostStoryRows,
  hasEvalData,
  summaryRows,
} from './eval-data.ts';

# Eval Summary

{!hasEvalData ? (
  <blockquote>{emptySummaryText}</blockquote>
) : (
  <>
    <table>
      <tbody>
        {summaryRows.map((row) => (
          <tr key={row.label}>
            <td><strong>{row.label}</strong></td>
            <td>
              {row.code ? (
                <code style={{ wordBreak: 'break-word' }}>{row.value}</code>
              ) : (
                row.value
              )}
            </td>
          </tr>
        ))}
        {ghostStoryRows.map((row) => (
          <tr key={row.label}>
            <td><strong>{row.label}</strong></td>
            <td>{row.value}</td>
          </tr>
        ))}
      </tbody>
    </table>

    ## Changed Files

    {changedFiles.length === 0 ? (
      <p>No file changes recorded.</p>
    ) : (
      <ul>
        {changedFiles.map((change) => (
          <li key={change.path}>
            <code>{change.gitStatus}</code> <code>{change.path}</code>
          </li>
        ))}
      </ul>
    )}
  </>
)}
